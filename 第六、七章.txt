#include<bits/stdc++.h>
void myswap(int *a, int *b){
    int tmp = *b;
    *b = *a;
    *a = tmp;
}
int main(){
    int n1,n2;
    std::cin >> n1 >> n2;;
	myswap(&n1,&n2);
	std::cout << n1 << " " << n2 << std::endl;
    return 0;
}

7.16
访问说明符出现的位置和次数没有限定。
对于我们愿意让整个程序都可以访问的成员应该定义在public之后。
对于我们只想让自己这个类单独使用的成员应该定义在private之后。
